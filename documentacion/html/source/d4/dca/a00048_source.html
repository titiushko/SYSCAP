278  public function validar_fechas($fecha1, $fecha2){
279  $fecha1 = empty($fecha1) ? '0000-00-00' : $fecha1;
280  $fecha2 = empty($fecha2) ? '0000-00-00' : $fecha2;
281  $valores_fecha1 = explode('-', $fecha1);
282  $valores_fecha2 = explode('-', $fecha2);
283  $dia1 = $valores_fecha1[2];
284  $mes1 = $valores_fecha1[1];
285  $anyo1 = $valores_fecha1[0];
286  $dia2 = $valores_fecha2[2];
287  $mes2 = $valores_fecha2[1];
288  $anyo2 = $valores_fecha2[0];
289  $dias_fecha1 = gregoriantojd($mes1, $dia1, $anyo1);
290  $dias_fecha2 = gregoriantojd($mes2, $dia2, $anyo2);
291  if($dias_fecha1 == $dias_fecha2){
292  $this->form_validation->set_message('validar_fechas', icono_notificacion('error').'El campo '.bold('%s').' no debe de ser igual al campo '.bold('Fecha 2').'.');
293  return FALSE;
294  }
295  elseif($dias_fecha1 > $dias_fecha2){
296  $this->form_validation->set_message('validar_fechas', icono_notificacion('error').'El campo '.bold('%s').' no puede ser mayor al campo '.bold('Fecha 2').'.');
297  return FALSE;
298  }
299  else{
300  return TRUE;
